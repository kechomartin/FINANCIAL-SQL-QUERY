Invoice & Sales Query Showcase

1. Monthly revenue & invoice count by product category
Beginner▼

Aggregates total sales and number of invoices per product category per month. Essential for category performance dashboards.

Concepts:GROUP BYDATE_TRUNCJOIN (items + invoices + products)
-- Monthly revenue by product category (last 12 months)
SELECT
  p.category,
  DATE_TRUNC('month', i.invoice_date) AS month,
  COUNT(DISTINCT i.invoice_id)       AS invoice_count,
  SUM(ii.quantity * ii.unit_price * (1 - ii.discount)) AS total_revenue,
  AVG(ii.quantity * ii.unit_price)        AS avg_line_value
FROM invoices i
JOIN invoice_items ii ON i.invoice_id = ii.invoice_id
JOIN products p       ON ii.product_id = p.product_id
WHERE i.invoice_date >= NOW() - INTERVAL '12 months'
  AND i.status = 'PAID'
GROUP BY p.category, month
ORDER BY month DESC, total_revenue DESC;

2. Top 10 customers by lifetime invoice value
Beginner▼

Identifies highest spending customers for VIP programs, personalized offers, and retention strategies.

Concepts:SUM & COUNTJOINORDER BY LIMIT
-- Top 10 customers by total paid invoice amount
SELECT
  c.customer_id,
  c.full_name,
  c.city,
  c.loyalty_tier,
  COUNT(i.invoice_id) AS total_invoices,
  SUM(i.total_amount) AS total_spent,
  AVG(i.total_amount) AS avg_invoice_value
FROM customers c
JOIN invoices i ON c.customer_id = i.customer_id
WHERE i.status = 'PAID'
GROUP BY c.customer_id, c.full_name, c.city, c.loyalty_tier
ORDER BY total_spent DESC
LIMIT 10;

3. Running invoice balance per customer over time
Intermediate▼

Cumulative spending per customer (running total of invoice amounts) ordered by date – great for understanding spending patterns and loyalty evolution.

Concepts:SUM() OVERPARTITION BYROWS UNBOUNDED PRECEDING
-- Running total spent per customer (in chronological order)
SELECT
  i.customer_id,
  c.full_name,
  i.invoice_date,
  i.invoice_id,
  i.total_amount,
  SUM(i.total_amount) OVER (
    PARTITION BY i.customer_id
    ORDER BY i.invoice_date
    ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
  ) AS running_spent_total
FROM invoices i
JOIN customers c ON i.customer_id = c.customer_id
WHERE i.status = 'PAID'
ORDER BY i.customer_id, i.invoice_date;

4. Discount segmentation using CTE — high discount vs low discount invoices
Intermediate▼

Categorises invoices based on average discount percentage and calculates total revenue impact — helps monitor margin erosion.

Concepts:WITH CTECASE WHENGROUP BY
-- Segment invoices by discount intensity
WITH invoice_discounts AS (
  SELECT
    i.invoice_id,
    i.customer_id,
    i.total_amount,
    AVG(ii.discount) AS avg_discount_pct,
    CASE
      WHEN AVG(ii.discount) = 0       THEN 'No discount'
      WHEN AVG(ii.discount) < 0.1   THEN 'Low discount'
      WHEN AVG(ii.discount) < 0.25  THEN 'Medium discount'
      ELSE                           'Heavy discount'
    END AS discount_segment
  FROM invoices i
  JOIN invoice_items ii ON i.invoice_id = ii.invoice_id
  GROUP BY i.invoice_id, i.customer_id, i.total_amount
)
SELECT
  discount_segment,
  COUNT(invoice_id) AS invoices,
  ROUND(SUM(total_amount),2) AS total_revenue,
  ROUND(AVG(total_amount),2) AS avg_invoice_size
FROM invoice_discounts
GROUP BY discount_segment
ORDER BY total_revenue DESC;

5. Anomaly detection — invoices > 3x customer's average invoice
Advanced▼

Flags invoices that are unusually large compared to a customer’s historical average spend — useful for identifying potential fraud or system errors.

Concepts:AVG() OVERCTEWindow function per customer
-- Flag abnormal invoice spikes per customer
WITH customer_avg AS (
  SELECT
    customer_id,
    invoice_id,
    invoice_date,
    total_amount,
    AVG(total_amount) OVER (
      PARTITION BY customer_id
    ) AS customer_lifetime_avg
  FROM invoices
  WHERE status = 'PAID'
)
SELECT
  ca.customer_id,
  c.full_name,
  ca.invoice_id,
  ca.invoice_date,
  ca.total_amount,
  ROUND(ca.customer_lifetime_avg, 2) AS avg_invoice,
  ROUND(ca.total_amount / NULLIF(ca.customer_lifetime_avg,0), 2) AS spike_ratio
FROM customer_avg ca
JOIN customers c ON ca.customer_id = c.customer_id
WHERE ca.total_amount > ca.customer_lifetime_avg * 3
ORDER BY spike_ratio DESC;

6. Customer LTV by signup cohort year
Advanced▼

Groups customers by their signup year and calculates average lifetime value (LTV) per cohort — essential for tracking customer quality over time.

Concepts:EXTRACTCohort analysisCTE + window
-- Customer LTV by signup year cohort
WITH customer_ltv AS (
  SELECT
    c.customer_id,
    c.full_name,
    EXTRACT(YEAR FROM c.signup_date) AS cohort_year,
    SUM(i.total_amount) AS lifetime_value,
    COUNT(i.invoice_id) AS total_orders
  FROM customers c
  LEFT JOIN invoices i ON c.customer_id = i.customer_id AND i.status = 'PAID'
  GROUP BY c.customer_id, c.full_name, cohort_year
),
cohort_stats AS (
  SELECT
    cohort_year,
    COUNT(customer_id) AS cohort_size,
    ROUND(AVG(lifetime_value), 2) AS avg_ltv,
    ROUND(AVG(total_orders), 1) AS avg_orders,
    SUM(lifetime_value) AS cohort_revenue
  FROM customer_ltv
  GROUP BY cohort_year
)
SELECT *,
  ROUND(100.0 * cohort_revenue / SUM(cohort_revenue) OVER(), 1) AS revenue_share_pct
FROM cohort_stats
ORDER BY cohort_year;
