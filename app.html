Query showcase

1. Monthly transaction volume by account type
Beginner▼

Aggregates total transaction volume and count grouped by account type and month — a common reporting query for operations dashboards.

Concepts:GROUP BYDATE_TRUNCJOIN
-- Monthly transaction volume per account type
SELECT
  a.account_type,
  DATE_TRUNC('month', t.txn_date) AS month,
  COUNT(t.txn_id)             AS txn_count,
  SUM(t.amount)               AS total_volume,
  AVG(t.amount)               AS avg_txn_size
FROM transactions t
JOIN accounts a ON t.account_id = a.account_id
WHERE t.txn_date >= NOW() - INTERVAL '12 months'
GROUP BY a.account_type, month
ORDER BY month DESC, total_volume DESC;

2. Top customers by total deposits
Beginner▼

Ranks customers by their total deposit activity, useful for identifying high-value clients for relationship management or upselling.

Concepts:JOINWHERE filterORDER BYLIMIT
-- Top 10 customers by total deposits
SELECT
  c.customer_id,
  c.full_name,
  c.country,
  COUNT(DISTINCT a.account_id) AS num_accounts,
  SUM(t.amount)                AS total_deposits
FROM customers c
JOIN accounts a    ON c.customer_id = a.customer_id
JOIN transactions t ON a.account_id  = t.account_id
WHERE t.txn_type = 'DEPOSIT'
  AND a.status    = 'ACTIVE'
GROUP BY c.customer_id, c.full_name, c.country
ORDER BY total_deposits DESC
LIMIT 10;

3. Running balance per account using window functions
Intermediate▼

Calculates a running account balance over time by accumulating transaction amounts in chronological order — a classic use of window functions in finance.

Concepts:SUM() OVERPARTITION BYORDER BY (window)
-- Running balance for each account over time
SELECT
  account_id,
  txn_date,
  txn_type,
  amount,
  SUM(
    CASE
      WHEN txn_type = 'DEPOSIT'  THEN  amount
      WHEN txn_type = 'WITHDRAW' THEN -amount
      ELSE 0
    END
  ) OVER (
    PARTITION BY account_id
    ORDER BY txn_date
    ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
  ) AS running_balance
FROM transactions
ORDER BY account_id, txn_date;

4. Loan default risk segmentation using CTEs
Intermediate▼

Uses a CTE to segment customers by their loan status and risk score, then produces a summary count per segment — useful for credit risk reporting.

Concepts:WITH (CTE)CASE WHENGROUP BY
-- Segment customers by loan risk profile
WITH customer_risk AS (
  SELECT
    c.customer_id,
    c.full_name,
    c.risk_score,
    l.status      AS loan_status,
    l.loan_amount,
    CASE
      WHEN c.risk_score < 0.3 THEN 'Low risk'
      WHEN c.risk_score < 0.7 THEN 'Medium risk'
      ELSE                       'High risk'
    END AS risk_band
  FROM customers c
  JOIN loans l ON c.customer_id = l.customer_id
  WHERE l.status IN ('ACTIVE', 'DEFAULTED')
)
SELECT
  risk_band,
  loan_status,
  COUNT(*)             AS customers,
  SUM(loan_amount)     AS total_exposure,
  AVG(risk_score)      AS avg_risk_score
FROM customer_risk
GROUP BY risk_band, loan_status
ORDER BY risk_band, loan_status;

5. Fraud detection — unusual transaction spikes
Advanced▼

Flags accounts where a single transaction is more than 3x the account's own average transaction size — a simple but realistic fraud signal using window functions.

Concepts:AVG() OVERCTEHAVINGSubquery filter
-- Flag transactions 3x above the account average
WITH account_stats AS (
  SELECT
    account_id,
    txn_id,
    txn_date,
    amount,
    merchant,
    AVG(amount) OVER (
      PARTITION BY account_id
    ) AS avg_txn_amount
  FROM transactions
  WHERE txn_type = 'DEBIT'
)
SELECT
  a.account_id,
  c.full_name,
  s.txn_date,
  s.merchant,
  s.amount,
  ROUND(s.avg_txn_amount, 2) AS account_avg,
  ROUND(s.amount / s.avg_txn_amount, 2) AS spike_ratio
FROM account_stats s
JOIN accounts a   ON s.account_id  = a.account_id
JOIN customers c  ON a.customer_id = c.customer_id
WHERE s.amount > s.avg_txn_amount * 3
ORDER BY spike_ratio DESC;

6. Customer lifetime value with cohort grouping
Advanced▼

Groups customers into join-year cohorts and calculates average lifetime transaction value per cohort — demonstrates cohort analysis, a key skill in fintech analytics.

Concepts:EXTRACTCohort analysisMulti-CTENTILE
-- Customer lifetime value by join-year cohort
WITH customer_totals AS (
  SELECT
    c.customer_id,
    c.full_name,
    EXTRACT(YEAR FROM c.joined_date) AS cohort_year,
    SUM(t.amount)                       AS lifetime_value,
    COUNT(t.txn_id)                     AS total_txns
  FROM customers c
  JOIN accounts a     ON c.customer_id = a.customer_id
  JOIN transactions t ON a.account_id  = t.account_id
  GROUP BY c.customer_id, c.full_name, cohort_year
),
cohort_summary AS (
  SELECT
    cohort_year,
    COUNT(customer_id)        AS cohort_size,
    ROUND(AVG(lifetime_value), 2) AS avg_ltv,
    ROUND(AVG(total_txns), 1)    AS avg_txns,
    SUM(lifetime_value)       AS cohort_revenue
  FROM customer_totals
  GROUP BY cohort_year
)
SELECT *,
  ROUND(cohort_revenue * 100.0 /
    SUM(cohort_revenue) OVER(), 1) AS revenue_share_pct
FROM cohort_summary
ORDER BY cohort_year;
